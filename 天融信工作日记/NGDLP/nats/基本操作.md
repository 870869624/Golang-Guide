# 订阅所有主题（通配符 > 匹配所有层级）
# 用于监听服务器上的所有消息流，常用于调试和监控
```bash
nats sub ">" --server nats://10.8.27.151:30422
```

# 订阅队列组 queue.ngdlp.sensitivefileupdatetopic
# 队列订阅实现负载均衡，消息只会被组内一个消费者接收
```bash
nats sub queue.ngdlp.sensitivefileupdatetopic --server nats://10.8.27.151:30422
```

# 列出所有 JetStream 流（Stream）
# 查看服务器上配置的持久化消息流列表
```bash
nats stream list --server nats://10.8.227.180:30422
```

# 列出 aiop 流下的所有消费者（Consumer）
# 查看指定消息流的消费者配置和消费进度
```bash
nats consumer list aiop --server nats://10.8.27.151:30422
```

# 订阅 aiop.ngdlp.sensitivefileupdatetopic 主题
# 直接订阅指定主题，接收该主题的实时消息
```bash
nats sub aiop.ngdlp.sensitivefileupdatetopic --server nats://10.8.27.151:30422
```

# 订阅 ngdlp.sensitivefileupdatetopic 主题
# 与上一条类似，订阅更短路径的同名主题
```bash
nats sub ngdlp.sensitivefileupdatetopic --server nats://10.8.27.151:30422
```

# 列出所有 JetStream 流（Stream）
# 查看服务器上配置的持久化消息流列表（IP 已脱敏）
```bash
nats stream list --server nats://10.8.227.180:30422
```

# 查看当前账户信息
# 显示账户的流数量、消费限制、存储配额等统计信息
```bash
nats account info --server nats://10.8.227.180:30422
```

# 向 test 主题发布消息 "hello"
# 简单的消息发布测试，验证发布功能是否正常
```bash
nats pub test "hello" --server nats://10.8.227.180:30422
```

# 查看 aiop 流中 ngdlp.sensitivefileupdatetopic 主题的消息
# 浏览流中特定主题的历史消息，不消费只查看
```bash
nats stream view aiop --subject ngdlp.sensitivefileupdatetopic --server nats://10.8.27.151:30422
```

# 查看 aiop 流的详细信息
# 显示流的配置、状态、消息数量、存储信息等
```bash
nats stream info aiop --server nats://10.8.227.180:30422
```
